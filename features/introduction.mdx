<Card title="Subdomain Monitoring" icon="globe" href="/features/asm/subdomains">

</Card>

<Card title="Exposure Detection Engine" icon="terminal" href="/features/asm/exposures">

</Card>

<Card title="YAML Rule Engine" icon="laptop-code" href="/features/asm/rules">

</Card>

<Card title="Scans & Scheduling" icon="magnifying-glass" href="/features/asm/scans">

</Card>

<Card title="Automated Reporting" icon="file-text" href="/features/asm/reports">
